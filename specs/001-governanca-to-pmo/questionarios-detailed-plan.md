function validarResposta(questao: Questao, resposta: any): ValidacaoResult {
  const erros: string[] = [];
  
  // Obrigatoriedade
  if (questao.obrigatoria && (!resposta || resposta === '' || 
      (Array.isArray(resposta) && resposta.length === 0))) {
    erros.push('Esta questÃ£o Ã© obrigatÃ³ria');
  }
  
  // ValidaÃ§Ãµes especÃ­ficas por tipo
  if (resposta) {
    switch (questao.tipo) {
      case 'texto_livre':
      case 'texto_curto':
        if (questao.validacao?.min_caracteres && resposta.length < questao.validacao.min_caracteres) {
          erros.push(`MÃ­nimo de ${questao.validacao.min_caracteres} caracteres`);
        }
        if (questao.validacao?.max_caracteres && resposta.length > questao.validacao.max_caracteres) {
          erros.push(`MÃ¡ximo de ${questao.validacao.max_caracteres} caracteres`);
        }
        if (questao.validacao?.padrao_regex) {
          const regex = new RegExp(questao.validacao.padrao_regex);
          if (!regex.test(resposta)) {
            erros.push(questao.validacao.mensagem_erro || 'Formato invÃ¡lido');
          }
        }
        break;
        
      case 'numero':
        const num = Number(resposta);
        if (questao.validacao?.min_valor && num < questao.validacao.min_valor) {
          erros.push(`Valor mÃ­nimo: ${questao.validacao.min_valor}`);
        }
        if (questao.validacao?.max_valor && num > questao.validacao.max_valor) {
          erros.push(`Valor mÃ¡ximo: ${questao.validacao.max_valor}`);
        }
        break;
        
      case 'upload':
        if (questao.validacao?.tamanho_max_mb) {
          const tamanhoMB = resposta.tamanho / (1024 * 1024);
          if (tamanhoMB > questao.validacao.tamanho_max_mb) {
            erros.push(`Tamanho mÃ¡ximo: ${questao.validacao.tamanho_max_mb} MB`);
          }
        }
        break;
    }
  }
  
  return {
    valida: erros.length === 0,
    erros
  };
}

async function calcularEstatisticas(questionario_id: string): Promise<AnaliseEstatistica> {
  const questionario = await db.questionarios.findById(questionario_id);
  const respostas = await db.respostas_questionario.findByQuestionario(questionario_id);
  
  const analise: AnaliseEstatistica = {
    questionario_id,
    total_respostas: respostas.length,
    respostas_completas: respostas.filter(r => r.status === 'completa').length,
    respostas_parciais: respostas.filter(r => r.status === 'em_progresso').length,
    taxa_conclusao: 0,
    tempo_medio_preenchimento_minutos: 0,
    analise_por_questao: []
  };
  
  analise.taxa_conclusao = (analise.respostas_completas / analise.total_respostas) * 100;
  
  const tempos = respostas
    .filter(r => r.metadata?.tempo_preenchimento_segundos)
    .map(r => r.metadata.tempo_preenchimento_segundos / 60);
  analise.tempo_medio_preenchimento_minutos = 
    tempos.length > 0 ? tempos.reduce((a, b) => a + b, 0) / tempos.length : 0;
  
  // Analisar cada questÃ£o
  for (const questao of questionario.questoes) {
    const respostasQuestao = respostas
      .map(r => r.respostas.find(resp => resp.questao_id === questao.id))
      .filter(r => r !== undefined);
    
    const analiseQuestao: any = {
      questao_id: questao.id,
      questao_ordem: questao.ordem,
      questao_pergunta: questao.pergunta,
      tipo: questao.tipo,
      total_respostas: respostasQuestao.length
    };
    
    switch (questao.tipo) {
      case 'multipla_escolha':
      case 'multipla_escolha_multipla':
        // Contar frequÃªncia de cada opÃ§Ã£o
        const contagem: { [opcao: string]: number } = {};
        respostasQuestao.forEach(resp => {
          const valores = Array.isArray(resp.valor) ? resp.valor : [resp.valor];
          valores.forEach(val => {
            contagem[val] = (contagem[val] || 0) + 1;
          });
        });
        
        analiseQuestao.distribuicao_opcoes = questao.opcoes.map(opcao => ({
          opcao_id: opcao.id,
          opcao_label: opcao.label,
          count: contagem[opcao.valor] || 0,
          percentual: (contagem[opcao.valor] || 0) / analiseQuestao.total_respostas * 100
        }));
        break;
        
      case 'escala':
        const valores = respostasQuestao.map(r => Number(r.valor)).filter(v => !isNaN(v));
        if (valores.length > 0) {
          valores.sort((a, b) => a - b);
          analiseQuestao.estatisticas_numericas = {
            media: valores.reduce((a, b) => a + b, 0) / valores.length,
            mediana: valores[Math.floor(valores.length / 2)],
            moda: calcularModa(valores),

          };
        }
        break;
        
      case 'texto_livre':
      case 'texto_curto':
        const textos = respostasQuestao.map(r => r.valor || '').filter(t => t.length > 0);
        const todasPalavras = textos.flatMap(t => t.toLowerCase().split(/\s+/));
        const palavrasUnicas = new Set(todasPalavras);
        
        const frequenciaPalavras: { [palavra: string]: number } = {};

        });
        
        analiseQuestao.analise_texto = {

          palavras_unicas: palavrasUnicas.size,

        }

    analise.analise_por_questao.push(analiseQuestao);
